 - Brute Force
```cpp
class Solution {
public:
    string reverseParentheses(string s)
    {
        stack<int> stk;
        string res;

        for (char c : s)
        {
            if (c == '(')
            {
                stk.push(res.size());
            }
            else if (c == ')')
            {
                int start = stk.top(); stk.pop();
                reverse(res.begin() + start, res.end());
            }
            else
            {
                res += c;
            }
        }
        return res;
    }
};
```
- T: $O(n^2)$
- S: $O(n)$
