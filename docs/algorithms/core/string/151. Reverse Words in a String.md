:::note - Use >>
```cpp
class Solution {
public:
    string reverseWords(string s)
    {
        stringstream ss(s);
        string token;
        s = "";
        while (ss >> token)
        {
            s = token + ' ' + s;
        }
        return s.substr(0, s.size() - 1);
    }
};
```

:::note - Use getline
```cpp
class Solution {
public:
    string reverseWords(string s)
    {
        stringstream ss(s);
        s = "";
        string token;
        while(getline(ss, token, ' '))
        {
            if(token.empty()) continue;
            if(s.empty())
            {
                s = token;
            }
            else
            {
                s = token + ' ' + s;
            }
        }
        return s;
    }
};
```
- T: $O(N)$
- S: $O(N)$
