```cpp
class Solution {
public:
    int removeDuplicates(vector<int>& nums)
    {
        int insertIndex = 1;
        int cnt = 1;

        for (int i = 1; i < nums.size(); ++i)
        {
            if (nums[i] == nums[i - 1]) ++cnt;
            else cnt = 1;
            if (cnt <= 2) nums[insertIndex++] = nums[i];
        }
        return insertIndex;
    }
};
```
- T: $O(N)$
- S: $O(1)$
